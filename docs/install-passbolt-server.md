<#
.SYNOPSIS
	Installs passbolt server
.DESCRIPTION
	This PowerShell script installs Passbolt Community Edition (CE) - Passbolt is a hybrid credential platform.
.EXAMPLE
	PS> ./install-passbolt-server.ps1
.LINK
	https://github.com/fleschutz/PowerShell
.NOTES
	Author: Markus Fleschutz | License: CC0
#>

#requires -version 5.1

try {
	$stopWatch = [system.diagnostics.stopwatch]::startNew()

	if ($IsLinux) {
		"⏳ (1/4) Loading passbolt-repo-setup.ce.sh..."
		& curl -LO "https://download.passbolt.com/ce/installer/passbolt-repo-setup.ce.sh"

		"⏳ (2/4) Loading passbolt-ce-SHA512SUM.txt..."
		& curl -LO "https://github.com/passbolt/passbolt-dep-scripts/releases/latest/download/passbolt-ce-SHA512SUM.txt"

		"⏳ (3/4) Verifying checksum and performing setup..."
		& sha512sum -c passbolt-ce-SHA512SUM.txt && sudo bash ./passbolt-repo-setup.ce.sh || echo "Bad checksum. Aborting" && rm -f passbolt-repo-setup.ce.sh

		"⏳ (4/4) Installing passbolt-server..."
		& sudo apt install passbolt-ce-server
	} else {
		throw "Sorry, only Linux installation currently supported"
	}
	[int]$elapsed = $stopWatch.Elapsed.TotalSeconds
	"✅ Passbolt server installed successfully in $($elapsed)s."
	exit 0 # success
} catch {
	"⚠️ ERROR: $($Error[0]) (script line $($_.InvocationInfo.ScriptLineNumber))"
	exit 1
}
