<#
.SYNOPSIS
	Pings a host
.DESCRIPTION
	This PowerShell script pings the given host.
.PARAMETER hostname
	Specifies the hostname or IP address to ping (x.com by default)
.EXAMPLE
	PS> ./ping-host.ps1 x.com
	‚úÖ Host 'x.com' with 20ms latency at IP 104.244.42.1 is up üëç
.LINK
	https://github.com/fleschutz/PowerShell
.NOTES
	Author: Markus Fleschutz | License: CC0
#>

param([string]$hostname = "x.com")

try {
	$remoteHosts = $hostname.Split(",")
	$tasks = $remoteHosts | foreach { (New-Object Net.NetworkInformation.Ping).SendPingAsync($_,5000) }
	[Threading.Tasks.Task]::WaitAll($tasks)
	foreach($ping in $tasks.Result) {
		if ($ping.Status -eq "Success") {
			Write-Output "‚úÖ Host '$hostname' with $($ping.RoundtripTime)ms latency at IP $($ping.Address) is up üëç"
			exit 0 # success
		} else {
			Write-Output "‚ö†Ô∏è No reply from '$hostname' (IP $($ping.Address)) - check the connection or maybe the host is down."
			exit 1
		}
	}
	Write-Output "‚ö†Ô∏è No reply from host '$hostname' - check the connection or maybe the host is down."
	exit 1
} catch {
	"‚ö†Ô∏è Error in line $($_.InvocationInfo.ScriptLineNumber): $($Error[0])"
	exit 1
}
