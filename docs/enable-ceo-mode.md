	"⏳ Open VPN settings..."
	& "$PSScriptRoot/open-vpn-settings.ps1"

	& "$PSScriptRoot/show-notification.ps1" "CEO mode enabled."
	"✅ CEO mode enabled."
	exit 0 # success
} catch {
	"⚠️ ERROR: $($Error[0]) (in line $($_.InvocationInfo.ScriptLineNumber))"
