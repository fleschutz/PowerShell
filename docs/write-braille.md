try {
	if ($text -eq "" ) { $text = read-host "Enter text to write" }

	[char[]]$ArrayOfChars = $text.ToUpper()
	write-output ""
	for ($Row = 1; $Row -lt 4; $Row++) {
		$Line = ""
		foreach($Char in $ArrayOfChars) {
			$Line += BrailleChar $Char $Row
			$Line += "  "
		}
		write-output $Line
	}
	write-output ""
	exit 0 # success
