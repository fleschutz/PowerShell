try {
	if (-not(Test-Path "~/Dropbox" -pathType container)) {
		throw "No üìÇDropbox folder in your home directory - is Dropbox installed?"
	}
	$path = Resolve-Path "~/Dropbox"
	Set-Location "$path"
	$files = Get-ChildItem $path -attributes !Directory
	$folders = Get-ChildItem $path -attributes Directory
	"üìÇ$path entered (has $($files.Count) files and $($folders.Count) subfolders)"
	exit 0 # success
