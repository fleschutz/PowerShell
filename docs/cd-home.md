	if (-not(Test-Path "~" -pathType container)) { throw "The path to home directory '$path' doesn't exist (yet)" }
	$path = Resolve-Path "~"
	Set-Location "$path"
	$files = Get-ChildItem $path -attributes !Directory
	$folders = Get-ChildItem $path -attributes Directory
